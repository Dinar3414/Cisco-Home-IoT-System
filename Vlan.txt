switch which connect with 
server via fa0/1,
Front switch via fa0/3
garage switch via fa0/7
kitchen switch via fa0/6
bedroom switch via fa0/5
living room switch via fa0/4
via access smart phone switch fa0/2
main router via Gig0/1


#Main Switch CLI

(
enable
configure terminal

vlan 10
 name FRONT
vlan 20
 name LIVING
vlan 30
 name BEDROOM
vlan 40
 name KITCHEN
vlan 50
 name GARAGE
vlan 60
 name SMARTPHONE
vlan 99
 name SERVER

exit


#Access Port server

interface fa0/1
 switchport mode access
 switchport access vlan 99
 spanning-tree portfast
 exit


#Switch to switch Trunk

interface range fa0/2 - 7
 switchport mode trunk
 switchport trunk allowed vlan 10,20,30,40,50,60,99
 exit

#Router Up Link
interface gig0/1
 switchport mode trunk
 switchport trunk allowed vlan 10,20,30,40,50,60,99
 exit


#save

end
write memory

)
